// PARSE FONKSİYONLARI
// ═══════════════════════════════════════════════════════
function parseKablo(txt) {
  const raw        = String(txt || '').replace(/\s+/g, ' ').trim();
  const damarSayi  = Number(raw.match(/^(\d)/)?.[1] || 3);
  const damar      = damarSayi + 'x';
  const kesitMatch = raw.match(/\d+x\s*([0-9]+(?:[.,][0-9]+)?)\s*(?:\+\s*([0-9]+(?:[.,][0-9]+)?))?\s*mm2/i);
  const kesit      = (kesitMatch?.[1] || '2,5').replace('.', ',');
  const toprak     = (kesitMatch?.[2] || '').replace('.', ',');
  let tip = 'NYY';
  for (const t of KABLO_TIP_LIST) {
    if (new RegExp('\\b' + t + '\\b', 'i').test(raw)) { tip = t; break; }
  }
  const metraj = Number(raw.match(/(\d+)\s*m\b/i)?.[1] || 10);
  return { damar, damar_sayi: damarSayi, kesit, toprak_kesit: toprak, tip, metraj, ham: raw };
}

function parseYuk(txt) {
  if (!txt || !txt.trim()) return null;
  const raw   = String(txt).trim();
  const parts = raw.split('-').map(s => s.trim());
  return { tip: parts[0] || '', faz: parts[1] || '', toplam_guc: Number(raw.match(/(\d+)\s*W/i)?.[1] || 0), ham: raw };
}

function parseSigorta(txt) {
  if (!txt || !txt.trim()) return null;
  const raw  = String(txt).trim();
  const faz  = raw.includes('3x') ? '3x' : '1x';
  const akim = Number(raw.match(/(\d+)\s*A/i)?.[1] || 0);
  return { faz, akim, ham: raw };
}

function parseRole(txtA, txtB) {
  if (!txtA && !txtB) return null;
  const rawA = String(txtA || '').trim();
  const rawB = String(txtB || '').trim();
  const tip  = rawA.includes('30mA') ? '30mA' : '300mA';
  const fazB = rawB.includes('4x') ? '4x' : '2x';
  const akim = Number(rawB.match(/(\d+)\s*A/i)?.[1] || 0);
  return { tip, faz: fazB, akim, hamA: rawA, hamB: rawB };
}

function parseKofre(txt) {
  if (!txt || !txt.trim()) return null;
  const raw   = String(txt).trim();
  const tip   = raw.includes('TMŞ') ? 'TMŞ' : 'YSYA';
  const deger = Number(raw.match(/(\d+)/)?.[1] || 0);
  return { tip, deger, ham: raw };
}

function parseSayac(tipTxt, trafoTxt) {
  const rawTip    = String(tipTxt  || '').trim();
  const rawTrafo  = String(trafoTxt || '').trim();
  const ilkKelime = rawTip.split(' ')[0];
  let sayacTipi = 'Trifaze';
  if (ilkKelime === '1x10')      sayacTipi = 'Monofaze';
  else if (ilkKelime === '3x10') sayacTipi = 'Trifaze';
  else if (ilkKelime === 'X5')   sayacTipi = 'X5';

  let trafoAkim   = 0;
  let trafoCarpan = 0;
  let trafoTam    = '';
  if (rawTrafo && rawTrafo.includes('/')) {
    const parts = rawTrafo.split('/');
    trafoAkim   = Number(parts[0]) || 0;
    trafoCarpan = Number(parts[1]) || 0;
    trafoTam    = rawTrafo;
  } else if (rawTrafo && /^\d+$/.test(rawTrafo)) {
    // geriye uyumluluk: eski kayıtlar sadece sayı içeriyorsa "/5" paydası varsayılır
    // (sistemde kullanılan tek standart akım trafo paydası 5'tir)
    trafoAkim   = Number(rawTrafo) || 0;
    trafoCarpan = 5;
    trafoTam    = trafoAkim + '/5';
  }
  return {
    sayac_tipi  : sayacTipi,
    trafo_orani : trafoAkim,     // eski alan: sayısal pay (geriye uyumluluk)
    trafo_tam   : trafoTam,      // "100/5" formatı
    trafo_akim  : trafoAkim,     // 100
    trafo_carpan: trafoCarpan    // 5
  };
}

// ═══════════════════════════════════════════════════════
// KABLO DAMAR BUTON YÖNETİMİ
// ═══════════════════════════════════════════════════════
function guncelleDamarButonlari(seciliTip, tercihliDamar) {
  const tekDamarMi  = SADECE_TEK_DAMAR.includes(seciliTip);
  const damarBtnler = document.querySelectorAll('#kablo-damar button');
  damarBtnler.forEach(btn => btn.classList.remove('active'));
  damarBtnler.forEach(btn => {
    const val = btn.getAttribute('data-value');
    btn.disabled = tekDamarMi ? (val !== '1x') : (val === '1x');
  });
  let hedefDamar = tekDamarMi ? '1x' : (tercihliDamar && tercihliDamar !== '1x' ? tercihliDamar : '3x');
  damarBtnler.forEach(btn => {
    if (btn.getAttribute('data-value') === hedefDamar && !btn.disabled) btn.classList.add('active');
  });
}

function guncelleKabloKesitListesi(mevcutKesit) {
  const kesitSel  = qs('kablo-kesit');
  if (!kesitSel) return;
  const seciliTip  = qs('kablo-tip')?.value || 'NYY';
  const aktifDamar = document.querySelector('#kablo-damar button.active');
  const damarSayi  = aktifDamar ? aktifDamar.getAttribute('data-value').replace('x', '') : '3';
  const tipData    = window._AK ? window._AK[seciliTip] : null;
  const damarData  = tipData ? tipData[damarSayi] : null;
  const mevcutKesitler = damarData ? Object.keys(damarData) : [];
  kesitSel.innerHTML = '';
  if (mevcutKesitler.length === 0) return;
  mevcutKesitler.forEach(k => {
    const opt = document.createElement('option');
    opt.value = opt.textContent = String(k).replace('.', ',');
    opt.textContent += ' mm²';
    kesitSel.appendChild(opt);
  });
  if (mevcutKesit) {
    const normalized = String(mevcutKesit).replace('.', ',');
    const match = mevcutKesitler.find(k => String(k).replace('.', ',') === normalized);
    if (match) kesitSel.value = String(match).replace('.', ',');
  }
}

function fillKabloSelects(mevcutTip, mevcutDamar, mevcutKesit) {
  const tipSel        = qs('kablo-tip');
  const topraklamaSel = qs('kablo-topraklama-kesit');
  if (!tipSel) return;
  tipSel.innerHTML = '';
  KABLO_TIP_LIST.forEach(t => {
    const opt = document.createElement('option');
    opt.value = opt.textContent = t;
    tipSel.appendChild(opt);
  });
  if (mevcutTip) tipSel.value = mevcutTip;
  if (topraklamaSel) {
    topraklamaSel.innerHTML = '';
    const yokOpt = document.createElement('option');
    yokOpt.value = 'yok'; yokOpt.textContent = 'Yok';
    topraklamaSel.appendChild(yokOpt);
    KABLO_KESIT_LIST.forEach(k => {
      const opt = document.createElement('option');
      opt.value = k; opt.textContent = k + ' mm²';
      topraklamaSel.appendChild(opt);
    });
  }
  guncelleDamarButonlari(mevcutTip || 'NYY', mevcutDamar);
  guncelleKabloKesitListesi(mevcutKesit);
}

function hesaplaOtomatikTopraklamaKesit(kesitStr) {
  const floatList = KABLO_KESIT_LIST.map(k => parseFloat(k.replace(',', '.')));
  const S = parseFloat(String(kesitStr).replace(',', '.'));
  if (isNaN(S)) return kesitStr;
  if (S <= 16) return kesitStr;
  if (S <= 35) return '16';
  const yari = S / 2;
  let sonuc = floatList[floatList.length - 1];
  for (const f of floatList) {
    if (f >= yari) { sonuc = f; break; }
  }
  return String(sonuc).replace('.', ',');
}

function guncelleTopraklamaKesitDurumu(autoCalc) {
  const wrapper    = qs('topraklama-kesit-wrapper');
  const sel        = qs('kablo-topraklama-kesit');
  if (!wrapper || !sel) return;
  const aktifDamar = document.querySelector('#kablo-damar button.active');
  const damar      = aktifDamar?.getAttribute('data-value') || '3x';
  const kesitVal   = qs('kablo-kesit')?.value || '2,5';

  wrapper.style.display = '';
  if (damar === '1x') {
    sel.disabled = true;
    if (Array.from(sel.options).find(o => o.value === kesitVal)) sel.value = kesitVal;
  } else if (damar === '5x') {
    sel.disabled = true;
    const otomatik = hesaplaOtomatikTopraklamaKesit(kesitVal);
    if (Array.from(sel.options).find(o => o.value === otomatik)) sel.value = otomatik;
  } else if (damar === '2x' || damar === '4x') {
    sel.disabled = false;
    if (autoCalc && !sel.dataset.userOverride) {
      const otomatik = hesaplaOtomatikTopraklamaKesit(kesitVal);
      if (Array.from(sel.options).find(o => o.value === otomatik)) sel.value = otomatik;
    }
  } else {
    sel.disabled = false;
    sel.value = 'yok';
  }
}

function getClickXY(evtOrEl) {
  const wrap  = qs('wrap');
  const rect  = wrap ? wrap.getBoundingClientRect() : { left:0, top:0, width:window.innerWidth, height:window.innerHeight };
  const wrapW = wrap ? wrap.clientWidth  : rect.width;
  const wrapH = wrap ? wrap.clientHeight : rect.height;
  let viewX = rect.left + 20, viewY = rect.top + 20;
  if (evtOrEl && typeof evtOrEl.clientX === 'number') { viewX = evtOrEl.clientX; viewY = evtOrEl.clientY; }
  const rx = Math.max(0, Math.min(1, (viewX - rect.left) / (rect.width  || 1)));
  const ry = Math.max(0, Math.min(1, (viewY - rect.top)  / (rect.height || 1)));
  return { x: rx * wrapW, y: ry * wrapH, wrapW, wrapH };
}

function positionModal(popup, evtOrEl) {
  const { x, y, wrapW, wrapH } = getClickXY(evtOrEl);
  popup.style.visibility = 'hidden';
  popup.style.display    = 'block';
  const modalW = popup.offsetWidth  || 280;
  const modalH = popup.offsetHeight || 340;
  const margin = 8;
  const left   = Math.max(margin, Math.min(x + 12, wrapW - modalW - margin));
  const top    = Math.max(margin, Math.min(y + 12, wrapH - modalH - margin));
  popup.style.left       = `${left}px`;
  popup.style.top        = `${top}px`;
  popup.style.visibility = '';
}

// ═══════════════════════════════════════════════════════
// MODAL AÇ
// ═══════════════════════════════════════════════════════
function acModal(evt, options, cfg) {
  evt = evt || window.event || { clientX:0, clientY:0, stopPropagation(){}, target:null };
  if (typeof evt.stopPropagation === 'function') evt.stopPropagation();
  modalConfig = cfg || {};
  const panel = normPanelName(cfg?.panel);
  showPanel(panel);

  if (panel === 'kofre') {
    const kofreData = parseKofre(getText('txt_kofre'));
    const kofreTip  = kofreData?.tip || 'YSYA';
    if (kofreData) setActiveButton('#kofre-tip', kofreTip);
    const kofreList = kofreTip === 'TMŞ' ? TMS_LIST : YSYA_LIST;
    fillSelect(qs('kofre-deger'), kofreList, kofreData?.deger || kofreList[0]);
  } else if (panel === 'kablo') {
    const targetId = cfg?.targetId;
    let kabloData = { damar:'3x', damar_sayi:3, kesit:'2,5', tip:'NYY', metraj:10, toprak_kesit:'' };
    if (targetId) kabloData = parseKablo(getText(targetId));
    fillKabloSelects(kabloData.tip, kabloData.damar, kabloData.kesit);
    qs('kablo-m').value = kabloData.metraj;
    const topraklamaSel = qs('kablo-topraklama-kesit');
    if (topraklamaSel) {
      topraklamaSel.dataset.userOverride = '';
      topraklamaSel.value = kabloData.toprak_kesit || 'yok';
    }
    const needsAutoCalc = (kabloData.damar === '2x' || kabloData.damar === '4x') && !kabloData.toprak_kesit;
    guncelleTopraklamaKesitDurumu(needsAutoCalc);
    setupKolonSigortaEkleButton();
  } else if (panel === 'yuk') {
    temizleGeciciYukButonlari();
    setupYukPanelTipi();
    const targetId = cfg?.targetId;
    const cosqWrapper = qs('yuk-cosq-wrapper');
    if (cfg?.yukTipiSabit) {
      if (cosqWrapper) cosqWrapper.style.display = 'block';
      const cosqInput = qs('yuk-cosq');
      if (cosqInput) cosqInput.value = (window._istekCosQ != null) ? window._istekCosQ : 0.8;
    } else {
      if (cosqWrapper) cosqWrapper.style.display = 'none';
    }
    if (targetId) {
      const yukData = parseYuk(getText(targetId));
      if (yukData) {
        if (!cfg?.yukTipiSabit) setActiveButton('#yuk-tip', yukData.tip);
        setActiveButton('#yuk-faz', yukData.faz);
        qs('yuk-w').value = yukData.toplam_guc;
      }
    }
  } else if (panel === 'sigorta') {
    const targetId          = cfg?.targetId;
    const isKolonSig        = !!cfg?.isKolonSig;
    const isSayacCikisSig   = !!cfg?.isSayacCikisSig;
    const kesiciWrapper     = qs('kesici-tip-wrapper');
    if (kesiciWrapper) kesiciWrapper.style.display = isKolonSig ? 'block' : 'none';
    const muhurluWrapper    = qs('muhurlu-alan-wrapper');
    if (muhurluWrapper) {
      if (isSayacCikisSig) {
        muhurluWrapper.style.display = 'block';
        const savedMuhurlu = window._muhurluAlanda || 'evet';
        setActiveButton('#muhurlu-alan-sec', savedMuhurlu);
      } else {
        muhurluWrapper.style.display = 'none';
      }
    }
    let sigAkim = '';
    if (targetId) {
      const sigData = parseSigorta(getText(targetId));
      if (sigData) { setActiveButton('#sig-faz', sigData.faz); sigAkim = sigData.akim; }
      if (isKolonSig) {
        const kesiciTipi = isDomVisible('tms_kolon') ? 'TMŞ' : 'K.Otomat';
        setActiveButton('#kesici-tip', kesiciTipi);
      }
    }
    const kesiciTipiAktif = isKolonSig
      ? (document.querySelector('#kesici-tip button.active')?.getAttribute('data-value') || 'K.Otomat')
      : 'K.Otomat';
    const sigList = (isKolonSig && kesiciTipiAktif === 'TMŞ') ? TMS_LIST : K_OTOMAT_LIST;
    fillSelect(qs('sig-a'), sigList, sigAkim);
    setupSigortaKaldirButton();
  } else if (panel === 'sayac') {
    const sayacData = parseSayac(getText('txt_sayac_tip'), getText('txt_sayac_trafo'));
    if (sayacData) {
      setActiveButton('#sayac-tip', sayacData.sayac_tipi);
      const trafoSel = qs('sayac-trafo-select');
      if (trafoSel) {
        if (sayacData.sayac_tipi === 'X5') {
          trafoSel.disabled = false;
          let trafoVal = '';
          if (sayacData.trafo_tam && String(sayacData.trafo_tam).includes('/')) {
            trafoVal = String(sayacData.trafo_tam);
          } else if (sayacData.trafo_orani && /^\d+$/.test(String(sayacData.trafo_orani))) {
            trafoVal = String(sayacData.trafo_orani) + '/5';
          }
          trafoSel.value = trafoVal;
        } else {
          trafoSel.disabled = true;
          trafoSel.value = '';
        }
      }
    }
    guncelleSayacTrafoAlani();
    const tipTamMetin  = getText('txt_sayac_tip').trim();
    const olcumKelime  = tipTamMetin.split(' ')[1] || '';
    if (olcumKelime === 'Kombi')  setActiveButton('#sayac-olcum-ozellik', 'Kombi Sayaç');
    else if (olcumKelime === 'Aktif') setActiveButton('#sayac-olcum-ozellik', 'Aktif Sayaç');
  } else if (panel === 'role') {
    const targets  = cfg?.targets || {};
    const roleData = parseRole(getText(targets.roleA), getText(targets.roleB));
    if (roleData) {
      setActiveButton('#role-ma',  roleData.tip);
      setActiveButton('#role-faz', roleData.faz);
    }
    fillSelect(qs('role-a'), K_OTOMAT_LIST, roleData?.akim || K_OTOMAT_LIST[0]);
  }

  const popup = qs('proje-popup');
  positionModal(popup, evt);
  popup.style.display = 'block';
}

// ── SVG serileştirirken hit-layer onclick attribute'larını geçici kaldır,
//    serialize et, sonra geri koy → boyutu şişirme
function _serializeSvgTemiz(svgEl) {
  if (!svgEl) return '';
  // onclick attribute'u olan hit-layer'ları bul
  var hitLayers = svgEl.querySelectorAll('.hit-layer[onclick]');
  var yedekler = [];
  hitLayers.forEach(function(el) {
    yedekler.push({ el: el, onclick: el.getAttribute('onclick') });
    el.removeAttribute('onclick');
  });
  var str = '';
  try { str = new XMLSerializer().serializeToString(svgEl); } catch(e) {}
  // Geri koy
  yedekler.forEach(function(y) { y.el.setAttribute('onclick', y.onclick); });
  return str;
}

// ── Gerilim düşümü verilerini rapor sayfasına uygun formatta hazırla
//    collectGDData ve hesapla fonksiyonları gerilim_dusumu script'inde tanımlı,
//    burada sadece tablo satırlarını çıkarıyoruz.
function _gdTabloyadonustur(tip) {
  if (typeof collectGDData !== 'function') return null;
  try {
    var d = collectGDData(tip);
    if (!d) return null;

    var kK = d.kolonK, dK = d.dtK, yK = d.yukK, yP = d.yukPar;

    // Sutun tanımları (rapor sayfasının gdTabloOlustur'una uyumlu)
    var sutunlar = [
      { ad: 'Hat',      w: 55 },
      { ad: 'Kablo',    w: 90 },
      { ad: 'Uzunluk',  w: 55 },
      { ad: 'Güç (W)',  w: 55 },
      { ad: '%ΔU',      w: 50 }
    ];

    function fy(v) {
      if (v == null || isNaN(v)) return '—';
      return '%' + parseFloat(v.toFixed(2));
    }
    function hGD_lokal(kp, g) {
      if (!kp || !g) return 0;
      var k = parseFloat(String(kp.kesit || '0').replace(',', '.'));
      if (!k || !kp.kablo_gd_u) return 0;
      return (kp.kablo_gd_100 * kp.metraj * g) / (56 * k * Math.pow(kp.kablo_gd_u, 2));
    }
    function hGDy_lokal(kp, yp, g) {
      if (!kp || !yp || !g) return 0;
      var k = parseFloat(String(kp.kesit || '0').replace(',', '.'));
      if (!k || !yp.gd_u) return 0;
      return (yp.yuk_gd_100 * kp.metraj * g) / (56 * k * Math.pow(yp.gd_u, 2));
    }

    var e1 = hGD_lokal(kK, d.toplamGuc);
    var e2 = hGD_lokal(dK, d.dtGuc);
    var e3 = hGDy_lokal(yK, yP, yP ? yP.birim_guc : 0);
    var et = e1 + e2 + e3;

    var satirlar = [
      [
        'Kolon Kablo',
        kK ? kK.kablo_damarkesit + ' mm²' : '—',
        kK ? kK.metraj + ' m'             : '—',
        d.toplamGuc ? d.toplamGuc + ' W'  : '—',
        fy(e1)
      ],
      [
        'DT Kablo',
        dK ? dK.kablo_damarkesit + ' mm²' : '—',
        dK ? dK.metraj + ' m'             : '—',
        d.dtGuc ? d.dtGuc + ' W'          : '—',
        fy(e2)
      ],
      [
        'Yük Kablo',
        yK ? yK.kablo_damarkesit + ' mm²' : '—',
        yK ? yK.metraj + ' m'             : '—',
        yP ? yP.birim_guc + ' W'          : '—',
        fy(e3)
      ],
      [
        'TOPLAM', '—', '—', '—',
        fy(et) + (et < 1.5 ? ' ✓' : ' ✗')
      ]
    ];

    return { sutunlar: sutunlar, satirlar: satirlar };
  } catch(e) {
    console.warn('[_gdTabloyadonustur] hata:', e);
    return null;
  }
}

function _linyeListesiOlustur() {
  var liste = [];

  var mevcutGuc = numVal('guc-eski-kurulu');

    liste.push({
      tip:      parcaM[0] || 'Mevcut',
      faz:      parcaM[1] || 'Trifaz',
      guc:      mevcutGuc,
      sigorta:  getText('txt_sig_mevcut'),
      kablo:    getText('txt_linye_dt_mevcut'),
      dt:       getText('txt_dt_mevcut'),
      aciklama: yukMevcut
    });
  }
  var yukIstek  = getText('txt_yuk_istek');
  var parcaI    = yukIstek.split('-').map(function(s){ return s.trim(); });
  var istekGuc  = numVal('guc-ilave-kurulu');
  if (istekGuc > 0 || yukIstek) {
    liste.push({
      tip:      parcaI[0] || 'İstek',
      faz:      parcaI[1] || 'Trifaz',
      guc:      istekGuc,
      sigorta:  getText('txt_sig_istek'),
      kablo:    getText('txt_linye_dt_istek'),
      dt:       'İSTEK',
      aciklama: yukIstek
    });
  }
  return liste;
}
