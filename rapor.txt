// ═══════════════════════════════════════════════════════
// SVG → Yüksek Kalite PNG
// Vektörel SVG'yi canvas'a çizip dataURL döner.
// dpiMul: canvas ölçeği (3 = ~300dpi eşdeğeri)
// ═══════════════════════════════════════════════════════
function svgToPng(svgString, pxW, pxH, dpiMul) {
  return new Promise(function(resolve, reject) {
    dpiMul = dpiMul || 3;
    const blob = new Blob([svgString], {type:'image/svg+xml;charset=utf-8'});
    const url  = URL.createObjectURL(blob);
    const img  = new Image();
    img.onload = function() {
      const cw = (pxW || img.naturalWidth  || 800) * dpiMul;
      const ch = (pxH || img.naturalHeight || 600) * dpiMul;
      const canvas  = document.createElement('canvas');
      canvas.width  = cw;
      canvas.height = ch;
      const c2d = canvas.getContext('2d');
      c2d.fillStyle = '#ffffff';
      c2d.fillRect(0, 0, cw, ch);
      c2d.drawImage(img, 0, 0, cw, ch);
      URL.revokeObjectURL(url);
      resolve({ dataUrl: canvas.toDataURL('image/png'), w: cw, h: ch, dpiMul: dpiMul });
    };
    img.onerror = function(e) {
      URL.revokeObjectURL(url);
      reject(new Error('SVG yüklenemedi'));
    };
    img.src = url;
  });
}

// ═══════════════════════════════════════════════════════
// HTML → Yüksek Kalite PNG (html2canvas)
// htmlStr : outerHTML string
// renderW : sandbox genişliği (layout)
// dpiMul  : html2canvas scale parametresi
// ═══════════════════════════════════════════════════════
function htmlToPng(htmlStr, renderW, dpiMul) {
  return new Promise(function(resolve, reject) {
    dpiMul  = dpiMul  || 3;
    renderW = renderW || 1200;

    // Sandbox ayarla
    sandbox.style.width      = renderW + 'px';
    sandbox.style.background = '#ffffff';

    // HTML'i sandbox'a yerleştir
    // Güvenli: outerHTML ya da innerHTML olabilir
    // Eğer içerik bir element outerHTML'iyse direkt koy
    sandbox.innerHTML = htmlStr;

    // Render edilecek element
    const el = sandbox.firstElementChild || sandbox;

    // Stillerin yüklenmesini bekle
    setTimeout(function() {
      window.html2canvas(el, {
        backgroundColor : '#ffffff',
        scale           : dpiMul,
        useCORS         : true,
        allowTaint      : true,
        logging         : false,
        windowWidth     : renderW,
        // Ekran dışı sandbox için scrollX/Y
        scrollX         : 0,
        scrollY         : 0,
        x               : 0,
        y               : 0,
      }).then(function(canvas) {
        sandbox.innerHTML = '';
        resolve({
          dataUrl : canvas.toDataURL('image/png'),
          w       : canvas.width,
          h       : canvas.height,
          dpiMul  : dpiMul,
        });
      }).catch(function(e) {
        sandbox.innerHTML = '';
        reject(e);
      });
    }, 80); // kısa bekleme → layout tamamlanması için
  });
}

// ═══════════════════════════════════════════════════════
// PDF'E PNG EMBED
// pdfH    : sayfa yüksekliği (pt) — pdf-lib sol-alt origin için
// imgX, imgY_top: sol-üst köşe koordinatı (x soldan, y yukarıdan)
// imgW, imgH    : PDF üzerindeki boyut (pt)
// ═══════════════════════════════════════════════════════
async function embedPng(pdfDoc, page, pdfH, dataUrl, imgX, imgY_top, imgW, imgH) {
  const resp     = await fetch(dataUrl);
  const arrBuf   = await resp.arrayBuffer();
  const pngImage = await pdfDoc.embedPng(arrBuf);
  // pdf-lib: y = sol-altta origin → dönüştür
  const pdfY = pdfH - imgY_top - imgH;
  page.drawImage(pngImage, { x: imgX, y: pdfY, width: imgW, height: imgH });
}

// Akım trafo oranı listesi (büyüklük sırasıyla)
const TRAFO_ORANLARI = [100, 125, 150, 200, 250, 300, 400, 500, 600, 750, 1250];

// Standart topraklama kesit tablosu (IEC 60364 / yönetmelik)
const STANDART_KESITLER = [1, 1.5, 2.5, 4, 6, 10, 16, 25, 35, 50, 70, 95, 120, 150, 185, 240];

// Faz kesitine göre önerilen topraklama kesitini hesaplar
function odTopraklamaOnerilen(kesit) {
  if (kesit <= 16) return kesit;
  if (kesit <= 35) return 16;
  const ham = kesit / 2;
  return STANDART_KESITLER.find(function(s) { return s >= ham; }) || STANDART_KESITLER[STANDART_KESITLER.length - 1];
}

function odParseKablo(txt) {
  if (!txt || !txt.trim()) return null;
  const raw = String(txt).replace(/\s+/g, ' ').trim();
  const damarSayi = Number(raw.match(/^(\d)/)?.[1] || 3);
  const damar = String(damarSayi);
  const kesitMatch = raw.match(/\d+x\s*([0-9]+(?:[.,][0-9]+)?)\s*(?:\+\s*([0-9]+(?:[.,][0-9]+)?))?\s*mm2/i);
  const kesitStr = (kesitMatch?.[1] || '2.5').replace(',', '.');
  const kesit = parseFloat(kesitStr);
  const toprakStr = (kesitMatch?.[2] || '').replace(',', '.');
  const toprak = parseFloat(toprakStr) || 0;
  let tip = 'NYY';
  const TIPLER = ['NYFGBY','N2XH','NYM','TTR','NYAF','NYA','Alüminyum','NYY'];
  for (const t of TIPLER) {
    if (new RegExp('\\b' + t + '\\b', 'i').test(raw)) { tip = t; break; }
  }
  const fazText = raw.match(/(\d+)x/i)?.[1] || '3';
  const fazSayi = parseInt(fazText);
  // Monofaz: 2x+toprak(3 damar) veya 3x. Trifaz: 4x+toprak(5 damar) veya 5x
  const isTrifaz = fazSayi >= 4;
  const topraklamaOnerilen = odTopraklamaOnerilen(kesit);
  const kullanilan_kisa = fazSayi + 'x' + kesitStr + (toprakStr ? '+' + toprakStr : '') + ' mm²';
  return { damar, damar_sayi: damarSayi, kesit, toprak, topraklamaOnerilen, tip, ham: raw, kullanilan_kisa, trifaz: isTrifaz, monofaz: !isTrifaz };
}

function odParseSigorta(txt) {
  if (!txt || !txt.trim()) return null;
  const raw = String(txt).trim();
  const faz = raw.includes('3x') ? '3x' : '1x';
  const akim = Number(raw.match(/(\d+)\s*A/i)?.[1] || 0);
  const tip = raw.includes('TMŞ') ? 'TMŞ' : 'K.Otomat';
  const kullanilan_kisa = faz + akim + 'A';
  return { faz, akim, tip, ham: raw, kullanilan_kisa };
}

function odParseRole(txtA, txtB) {
  if (!txtA && !txtB) return null;
  const rawA = String(txtA || '').trim();
  const rawB = String(txtB || '').trim();
  const ma = rawA.includes('30mA') ? 30 : 300;
  const faz = rawB.includes('4x') ? '4x' : '2x';
  const akim = Number(rawB.match(/(\d+)\s*A/i)?.[1] || 0);
  const kullanilan_kisa = ma + 'mA / ' + faz + akim + 'A';
  return { ma, faz, akim, hamA: rawA, hamB: rawB, kullanilan_kisa };
}

function odParseSayac(tipTxt, trafoTxt) {
  const rawTip = String(tipTxt || '').trim();
  const rawTrafo = String(trafoTxt || '').trim();
  const ilkKelime = rawTip.split(' ')[0];
  let tip = 'Trifaze';
  if (ilkKelime === '1x10') tip = 'Monofaze';
  else if (ilkKelime === 'X5') tip = 'X5';
  const trafoTam = rawTrafo && rawTrafo.includes('/') ? rawTrafo : '';
  const parts = trafoTam ? trafoTam.split('/') : ['', ''];
  const trafoAkim = parseInt(parts[0]) || 0;
  const trafoCarpan = parseInt(parts[1]) || 5;
  return { tip, trafoTam, trafoAkim, trafoCarpan };
}

function odParseYuk(txt) {
  if (!txt || !txt.trim()) return null;
  const raw = String(txt).trim();
  const parts = raw.split('-').map(s => s.trim());
  const faz = parts[1] || '';
  const guc = Number(raw.match(/(\d+)\s*W/i)?.[1] || 0);
  const isTrifaz = /trifaz/i.test(faz) || /3x/i.test(faz);
  return { tip: parts[0] || '', faz, guc, trifaz: isTrifaz, monofaz: !isTrifaz, ham: raw };
}

function odGetAkimTasima(tip, damar, kesit) {
  const tipData = AK_TABLO[tip];
  if (!tipData) return null;
  const damarData = tipData[String(damar)];
  if (!damarData) return null;
  return damarData[kesit] || null;
}

// Verilen yukAkimi için uygun kablo kesitlerini bul
// Döner: kablo tipi ve damar için verilen yukAkiminden büyük ilk iki kesiti
function odUygunKesitler(tip, damar, yukAkim) {
  const tipData = AK_TABLO[tip];
  if (!tipData) return [];
  const damarData = tipData[String(damar)];
  if (!damarData) return [];
  const kesitler = Object.keys(damarData).map(Number).sort((a, b) => a - b);
  const uygunlar = kesitler.filter(k => damarData[k] >= yukAkim);
  return uygunlar; // İlk iki tanesi uygun
}

// Sigorta değerlendirmesi: seçilen akım için kategori belirle
// Döner: 'uygun', 'potansiyel', 'uygunsuz'
function odSigortaKategori(secilen, yukAkim, liste) {
  if (!secilen || !yukAkim) return 'uygunsuz';
  const sirali = liste.slice().sort((a, b) => a - b);
  // en_yakin_asagi: yukAkim'e eşit veya küçük en büyük sigorta
  const asagilar = sirali.filter(s => s <= yukAkim);
  const enYakinAsagi = asagilar.length > 0 ? asagilar[asagilar.length - 1] : null;

  // a1 koşulu: enYakinAsagi <= yukAkim <= enYakinAsagi * 1.05
  const a1 = enYakinAsagi !== null && yukAkim <= enYakinAsagi * 1.05;

  if (a1) {
    // enYakinAsagi indeksi
    const idx = sirali.indexOf(enYakinAsagi);
    const birUst  = sirali[idx + 1];
    const ikiUst  = sirali[idx + 2];
    if (secilen === enYakinAsagi) return 'potansiyel';
    if (birUst !== undefined && secilen === birUst) return 'uygun';
    if (ikiUst !== undefined && secilen === ikiUst) return 'potansiyel';
    return 'uygunsuz';
  } else {
    // a2: yukAkim > enYakinAsagi*1.05 veya yukAkim < en küçük sigorta
    // enYakinYukari: yukAkim'den büyük veya eşit en küçük sigorta
    const yukarilar = sirali.filter(s => s >= yukAkim);
    const enYakinYukari = yukarilar.length > 0 ? yukarilar[0] : null;
    if (!enYakinYukari) return 'uygunsuz';
    const idx = sirali.indexOf(enYakinYukari);
    const birUst = sirali[idx + 1];
    if (secilen === enYakinYukari) return 'uygun';
    if (birUst !== undefined && secilen === birUst) return 'potansiyel';
    return 'uygunsuz';
  }
}

function odDurumBadge(durum) {
  if (durum === 'uygun')     return '<span class="od-durum od-uygun">✅ UYGUN</span>';
  if (durum === 'potansiyel') return '<span class="od-durum od-potansiyel">⚠️ RİSK</span>';
  if (durum === 'atla')      return '<span class="od-durum od-atla">— PASİF</span>';
  return '<span class="od-durum od-uygunsuz">❌ UYGUNSUZ</span>';
}

}

// Verilen yukAkimi için uygun (uygun) sigorta değerini döndürür
function odUygunSigortaDeger(yukAkim, liste) {
  if (!yukAkim) return null;
  const sirali = liste.slice().sort(function(a, b) { return a - b; });
  const asagilar = sirali.filter(function(s) { return s <= yukAkim; });
  const enYakinAsagi = asagilar.length > 0 ? asagilar[asagilar.length - 1] : null;
  const a1 = enYakinAsagi !== null && yukAkim <= enYakinAsagi * 1.05;
  if (a1) {
    const idx = sirali.indexOf(enYakinAsagi);
    return sirali[idx + 1] !== undefined ? sirali[idx + 1] : enYakinAsagi;
  } else {
    const yukarilar = sirali.filter(function(s) { return s >= yukAkim; });
    return yukarilar.length > 0 ? yukarilar[0] : null;
  }
}

    const katIst = odSigortaKategori(sigIstek.akim, akimIst, K_OTOMAT_LIST);
    const uygunStrIst = uygunSigIst ? uygunSigIst + 'A' : '?';
    kural.push({no:24, grup:'2. KESİCİLER', baslik:'İstek Kolon Kablosu Sigortası Akım',
      durum: katIst,
      not: katIst === 'uygunsuz' ? 'Uygun değil (uygun değer: ' + uygunStrIst + ')'
        : (katIst === 'potansiyel' ? 'Riskli (uygun değer: ' + uygunStrIst + ')' : ''),
      kullanilan: sigIstek.kullanilan_kisa, onerilen_row: onRowIst, alan_adi: 'sigorta akım değeri'
    });
  }

  // ── 3. RÖLELER ───────────────────────────────────────

  // Kural 25-27: Mevcut röle
  if (roleMevcut) {
    const k25ok = sayacMonofaz ? roleMevcut.faz === '2x' : roleMevcut.faz === '4x';
    const beklenFazRoleMev = sayacMonofaz ? '2x' : '4x';
    const beklenFazStrRoleMev = sayacMonofaz ? '2x (monofaz)' : '4x (trifaz)';
    const akimRoleMev = sayacMonofaz ? yukAkimMonofaz(eskiTalep) : yukAkimTrifaz(eskiTalep);
    const uygunSigRoleMev = odUygunSigortaDeger(akimRoleMev, K_OTOMAT_LIST);
    const onRowRoleMev = '30mA / ' + beklenFazRoleMev + (uygunSigRoleMev || '?') + 'A';
    kural.push({no:25, grup:'3. RÖLELER', baslik:'Mevcut Röle Faz',
      durum: k25ok ? 'uygun' : 'uygunsuz',
      not: k25ok ? '' : 'Uygun değil (uygun değer: ' + beklenFazStrRoleMev + ')',
      kullanilan: roleMevcut.kullanilan_kisa, onerilen_row: onRowRoleMev, alan_adi: 'faz değeri'
    });
    const katRoleMev = odSigortaKategori(roleMevcut.akim, akimRoleMev, K_OTOMAT_LIST);
    const uygunStrRoleMev = uygunSigRoleMev ? uygunSigRoleMev + 'A' : '?';
    kural.push({no:26, grup:'3. RÖLELER', baslik:'Mevcut Röle Akım',
      durum: katRoleMev,
      not: katRoleMev === 'uygunsuz' ? 'Uygun değil (uygun değer: ' + uygunStrRoleMev + ')'
        : (katRoleMev === 'potansiyel' ? 'Riskli (uygun değer: ' + uygunStrRoleMev + ')' : ''),
      kullanilan: roleMevcut.kullanilan_kisa, onerilen_row: onRowRoleMev, alan_adi: 'röle akım değeri'
    });
    const k27ok = roleMevcut.ma === 30;
    kural.push({no:27, grup:'3. RÖLELER', baslik:'Mevcut Röle Tipi (30mA)',
      durum: k27ok ? 'uygun' : 'uygunsuz',
      not: k27ok ? '' : 'Uygun değil (uygun değer: 30mA)',
      kullanilan: roleMevcut.kullanilan_kisa, onerilen_row: onRowRoleMev, alan_adi: 'röle tipi'
    });
  }

  // Kural 28-30: İstek röle
  if (roleIstek) {
    let beklenFazRoleIst;
    if (sayacMonofaz) {
      beklenFazRoleIst = '2x';
    } else if (yukIstek) {
      beklenFazRoleIst = yukIstek.trifaz ? '4x' : '2x';
    } else {
      beklenFazRoleIst = '4x';
    }
    const k28ok = roleIstek.faz === beklenFazRoleIst;
    const beklenFazStrRoleIst = beklenFazRoleIst === '4x' ? '4x (trifaz)' : '2x (monofaz)';
    const istekTrifazRole = yukIstek ? yukIstek.trifaz : !sayacMonofaz;
    const akimRoleIst = istekTrifazRole ? yukAkimTrifaz(ilaveTalep) : yukAkimMonofaz(ilaveTalep);
    const uygunSigRoleIst = odUygunSigortaDeger(akimRoleIst, K_OTOMAT_LIST);
    const onRowRoleIst = '30mA / ' + beklenFazRoleIst + (uygunSigRoleIst || '?') + 'A';
    kural.push({no:28, grup:'3. RÖLELER', baslik:'İstek Röle Faz',
      durum: k28ok ? 'uygun' : 'uygunsuz',
      not: k28ok ? '' : 'Uygun değil (uygun değer: ' + beklenFazStrRoleIst + ')',
      kullanilan: roleIstek.kullanilan_kisa, onerilen_row: onRowRoleIst, alan_adi: 'faz değeri'
    });
    const katRoleIst = odSigortaKategori(roleIstek.akim, akimRoleIst, K_OTOMAT_LIST);
    const uygunStrRoleIst = uygunSigRoleIst ? uygunSigRoleIst + 'A' : '?';
    kural.push({no:29, grup:'3. RÖLELER', baslik:'İstek Röle Akım',
      durum: katRoleIst,
      not: katRoleIst === 'uygunsuz' ? 'Uygun değil (uygun değer: ' + uygunStrRoleIst + ')'
        : (katRoleIst === 'potansiyel' ? 'Riskli (uygun değer: ' + uygunStrRoleIst + ')' : ''),
      kullanilan: roleIstek.kullanilan_kisa, onerilen_row: onRowRoleIst, alan_adi: 'röle akım değeri'

      durum: k30ok ? 'uygun' : 'uygunsuz',
      not: k30ok ? '' : 'Uygun değil (uygun değer: 30mA)',
      kullanilan: roleIstek.kullanilan_kisa, onerilen_row: onRowRoleIst, alan_adi: 'röle tipi'
    });
  }

  // Kural 31-33: Sayaç giriş rölesi
  if (roleSayacGiris) {
    const beklenFazSGR = yeniKurulu > 5000 ? '4x' : '2x';
    const beklenFazStrSGR = yeniKurulu > 5000 ? '4x (trifaz)' : '2x (monofaz)';
    const k31ok = roleSayacGiris.faz === beklenFazSGR;
    const monofazMiSGR = yeniKurulu <= 5000;
    const akimSGR = monofazMiSGR ? yukAkimMonofaz(yeniTalep) : yukAkimTrifaz(yeniTalep);
    const uygunSigSGR = odUygunSigortaDeger(akimSGR, K_OTOMAT_LIST);
    const onRowSGR = '300mA / ' + beklenFazSGR + (uygunSigSGR || '?') + 'A';
    kural.push({no:31, grup:'3. RÖLELER', baslik:'Sayaç Giriş Rölesi Faz',
      durum: k31ok ? 'uygun' : 'uygunsuz',
      not: k31ok ? '' : 'Uygun değil (uygun değer: ' + beklenFazStrSGR + ')',
      kullanilan: roleSayacGiris.kullanilan_kisa, onerilen_row: onRowSGR, alan_adi: 'faz değeri'
    });
    const katSGR = odSigortaKategori(roleSayacGiris.akim, akimSGR, K_OTOMAT_LIST);
    const uygunStrSGR = uygunSigSGR ? uygunSigSGR + 'A' : '?';
    kural.push({no:32, grup:'3. RÖLELER', baslik:'Sayaç Giriş Rölesi Akım',
      durum: katSGR,
      not: katSGR === 'uygunsuz' ? 'Uygun değil (uygun değer: ' + uygunStrSGR + ')'
        : (katSGR === 'potansiyel' ? 'Riskli (uygun değer: ' + uygunStrSGR + ')' : ''),
      kullanilan: roleSayacGiris.kullanilan_kisa, onerilen_row: onRowSGR, alan_adi: 'röle akım değeri'
    });
    const k33ok = roleSayacGiris.ma === 300;
    kural.push({no:33, grup:'3. RÖLELER', baslik:'Sayaç Giriş Rölesi Tipi (300mA)',
      durum: k33ok ? 'uygun' : 'uygunsuz',
      not: k33ok ? '' : 'Uygun değil (uygun değer: 300mA)',
      kullanilan: roleSayacGiris.kullanilan_kisa, onerilen_row: onRowSGR, alan_adi: 'röle tipi'
    });
  }

  // ── 4. SAYAÇ ─────────────────────────────────────────

  // Kural 34: Sayaç tipi
  const akim34 = yukAkimTrifaz(yeniTalep);
  let beklenTip;
  if (yeniKurulu > 5000 && akim34 < 100) beklenTip = 'Trifaze';
  else if (yeniKurulu > 5000 && akim34 >= 100) beklenTip = 'X5';
  else beklenTip = 'Monofaze';
  {
    const k34ok = sayac.tip === beklenTip;
    kural.push({no:34, grup:'4. SAYAÇ', baslik:'Sayaç Tipi',
      durum: k34ok ? 'uygun' : 'uygunsuz',
      not: k34ok ? '' : 'Sayaç tipi yeni talep güç değerine göre UYGUN DEĞİLDİR. Projenin onaydan dönme riskine karşı değerleri kontrol etmeniz önerilir.',
      kullanilan: sayac.tip, onerilen_row: beklenTip, alan_adi: 'sayaç tipi'
    });
  }

  // Kural 35: Akım Trafo Oranı (sadece önerilen sayaç X5 ise)
  if (beklenTip === 'X5') {
    const yukAkim = yukAkimTrifaz(yeniTalep);
    const bulunanOran = yukAkim;
    const secilenAkim = sayac.trafoAkim;
    const sirali = TRAFO_ORANLARI.slice().sort(function(a, b) { return a - b; });
    const uygunlar = sirali.filter(function(s) { return s >= bulunanOran; });
    const ilkUygun = uygunlar.length > 0 ? uygunlar[0] : null;
    const uygunStr = ilkUygun ? ilkUygun + '/5' : '?';
    if (secilenAkim > 0) {
      const aPotansiyel = secilenAkim < bulunanOran && bulunanOran <= secilenAkim * 1.05;
      let kat;
      if (aPotansiyel) {
        kat = 'potansiyel';
      } else {
        if (!ilkUygun) {
          kat = 'uygunsuz';
        } else {
          const idx = sirali.indexOf(ilkUygun);
          const birUst = sirali[idx + 1];
          if (secilenAkim === ilkUygun) kat = 'uygun';
          else if (birUst !== undefined && secilenAkim === birUst) kat = 'potansiyel';
          else kat = 'uygunsuz';
        }
      }
      kural.push({no:35, grup:'4. SAYAÇ', baslik:'Akım Trafo Oranı',
        durum: kat,
        not: kat === 'uygunsuz' ? 'Uygun değil (uygun değer: ' + uygunStr + ')'
          : (kat === 'potansiyel' ? 'Riskli (uygun değer: ' + uygunStr + ')' : ''),
        kullanilan: sayac.trafoTam || '—', onerilen_row: uygunStr, alan_adi: 'akım trafo oranı'
      });
    } else {
      kural.push({no:35, grup:'4. SAYAÇ', baslik:'Akım Trafo Oranı',
        durum: 'uygunsuz',
        not: 'Akım trafo oranı seçilmemiş (uygun değer: ' + uygunStr + ')',
        kullanilan: '—', onerilen_row: uygunStr, alan_adi: 'akım trafo oranı'
      });
    }
  } else {
    kural.push({no:35, grup:'4. SAYAÇ', baslik:'Akım Trafo Oranı',
      durum:'atla', not:'Sayaç monofaze/trifaze — trafo oranı kuralı uygulanmaz',
      kullanilan: '—', onerilen_row: '—', alan_adi: 'akım trafo oranı'
    });
  }

  // ── 5. GERİLİM DÜŞÜMÜ HESAPLARI ─────────────────────

  // Kural 36: En uzun %etop < 3 olmalı
  {
    const gdUzun = sema.gdEnUzun;
    if (gdUzun && gdUzun.etf && gdUzun.etf !== '—') {
      const etop = (gdUzun.et !== undefined && gdUzun.et !== null) ? gdUzun.et : parseFloat(String(gdUzun.etf).replace(',', '.').replace('%', ''));
      const ok = !isNaN(etop) && etop < 3;
      kural.push({no:36, grup:'5. GERİLİM DÜŞÜMÜ', baslik:'Gerilim düşümü hesabı (en uzun) için gerilim düşümü uygunluğu',
        durum: ok ? 'uygun' : 'uygunsuz',
        not: ok ? '' : 'Gerilim düşümü hesabı (en uzun) için %etop değeri UYGUN DEĞİLDİR. Projenin onaylanmama riskine karşı değerleri kontrol etmeniz önerilir.',
        kullanilan: gdUzun.etf, onerilen_row: '<%3', alan_adi: '%etop değeri'
      });
    } else {
      kural.push({no:36, grup:'5. GERİLİM DÜŞÜMÜ', baslik:'Gerilim düşümü hesabı (en uzun) için gerilim düşümü uygunluğu',
        durum:'atla', not:'Gerilim düşümü hesabı verisi bulunamadı.',
        kullanilan: '—', onerilen_row: '—', alan_adi: '%etop değeri'
      });
    }
  }

  // Kural 37: En güçlü %etop < 3 olmalı
  {
    const gdGuclu = sema.gdEnGuclu;
    if (gdGuclu && gdGuclu.etf && gdGuclu.etf !== '—') {
      const etop = (gdGuclu.et !== undefined && gdGuclu.et !== null) ? gdGuclu.et : parseFloat(String(gdGuclu.etf).replace(',', '.').replace('%', ''));
      const ok = !isNaN(etop) && etop < 3;
      kural.push({no:37, grup:'5. GERİLİM DÜŞÜMÜ', baslik:'Gerilim düşümü hesabı (en güçlü) için gerilim düşümü uygunluğu',
        durum: ok ? 'uygun' : 'uygunsuz',
        not: ok ? '' : 'Gerilim düşümü hesabı (en güçlü) için %etop değeri UYGUN DEĞİLDİR. Projenin onaylanmama riskine karşı değerleri kontrol etmeniz önerilir.',
        kullanilan: gdGuclu.etf, onerilen_row: '<%3', alan_adi: '%etop değeri'
      });
    } else {
      kural.push({no:37, grup:'5. GERİLİM DÜŞÜMÜ', baslik:'Gerilim düşümü hesabı (en güçlü) için gerilim düşümü uygunluğu',
        durum:'atla', not:'Gerilim düşümü hesabı verisi bulunamadı.',
        kullanilan: '—', onerilen_row: '—', alan_adi: '%etop değeri'
      });
    }
  }

  // Kural 38: En uzun hat — I1, I2, I3 akım kablo uygunluğu
  {
    const gdUzun = sema.gdEnUzun;
    if (gdUzun && gdUzun.ah) {
      const akimlar = [
        {no:38,   noStr:'I1', val:gdUzun.F1,         n:gdUzun.n1},
        {no:38.1, noStr:'I2', val:gdUzun.ah.I2f,     n:gdUzun.n2},
        {no:38.2, noStr:'I3', val:gdUzun.ah.I3f,     n:gdUzun.n3}
      ];
      akimlar.forEach(function(a) {
        const ok = a.n ? a.n.ok : null;
        const nHtml = a.n ? ' <span style="color:' + (a.n.ok ? '#2e7d32' : '#c62828') + '">' + a.n.s1 + ' — ' + a.n.s2 + '</span>' : '';
        kural.push({no:a.no, grup:'6. AKIM HESABI',
          baslik:'Gerilim düşümü hesabı (en uzun) için akım hesabı ' + a.noStr + ' uygunluğu',
          durum: ok === null ? 'atla' : (ok ? 'uygun' : 'uygunsuz'),
          not: '',
          kullanilan: '', onerilen_row: '',
          custom_degerlendirme: (a.val || '—') + nHtml,
          alan_adi: a.noStr + ' akım değeri'

      });
    } else {
      kural.push({no:38, grup:'6. AKIM HESABI', baslik:'Gerilim düşümü hesabı (en uzun) için akım hesabı uygunluğu',
        durum:'atla', not:'Akım hesabı verisi bulunamadı.',
        kullanilan: '—', onerilen_row: '—', alan_adi: 'akım değeri'
      });
    }
  }

  // Kural 39: En güçlü hat — I1, I2, I3 akım kablo uygunluğu
  {
    const gdGuclu = sema.gdEnGuclu;
    if (gdGuclu && gdGuclu.ah) {
      const akimlar = [
        {no:39,   noStr:'I1', val:gdGuclu.F1,         n:gdGuclu.n1},
        {no:39.1, noStr:'I2', val:gdGuclu.ah.I2f,     n:gdGuclu.n2},
        {no:39.2, noStr:'I3', val:gdGuclu.ah.I3f,     n:gdGuclu.n3}
      ];
      akimlar.forEach(function(a) {
        const ok = a.n ? a.n.ok : null;
        const nHtml = a.n ? ' <span style="color:' + (a.n.ok ? '#2e7d32' : '#c62828') + '">' + a.n.s1 + ' — ' + a.n.s2 + '</span>' : '';
        kural.push({no:a.no, grup:'6. AKIM HESABI',
          baslik:'Gerilim düşümü hesabı (en güçlü) için akım hesabı ' + a.noStr + ' uygunluğu',
          durum: ok === null ? 'atla' : (ok ? 'uygun' : 'uygunsuz'),
          not: '',
          kullanilan: '', onerilen_row: '',
          custom_degerlendirme: (a.val || '—') + nHtml,
          alan_adi: a.noStr + ' akım değeri'
        });
      });
    } else {
      kural.push({no:39, grup:'6. AKIM HESABI', baslik:'Gerilim düşümü hesabı (en güçlü) için akım hesabı uygunluğu',
        durum:'atla', not:'Akım hesabı verisi bulunamadı.',
        kullanilan: '—', onerilen_row: '—', alan_adi: 'akım değeri'
      });
    }
  }

  // ── HTML OLUŞTUR ──────────────────────────────────────
  const gruplar = {};
  kural.forEach(function(k) {
    if (!gruplar[k.grup]) gruplar[k.grup] = [];
    gruplar[k.grup].push(k);
  });

  // Başlıktaki son ekleri kaldırarak temel başlık döndürür
  function odBaslikGrubu(baslik) {
    return baslik.replace(/\s+(Topraklama\s+Kesit|Faz|Kesit|Akım|Varlığı|Durumu|Tipi\s*\(\d+mA\)|Tipi)$/, '').trim();
  }

  // Değerlendirme hücresi HTML'i oluşturur
  function odDegerlendirmeHtml(grupAdi, bg, list) {
    const allAtla = list.every(function(k) { return k.durum === 'atla'; });
    if (allAtla) return '<span style="color:#888">— PASİF</span>';
    const uygunsuzlar = list.filter(function(k) { return k.durum === 'uygunsuz'; });
    const potansiyellar = list.filter(function(k) { return k.durum === 'potansiyel'; });
    if (uygunsuzlar.length === 0 && potansiyellar.length === 0) {
      return '<span style="color:#2e7d32;font-weight:600">✅ Uygun</span>';
    }
    const parcalar = [];
    if (uygunsuzlar.length > 0) {
      const alanlar = uygunsuzlar.map(function(k) { return k.alan_adi || ''; }).filter(Boolean);
      parcalar.push('<span style="color:#c62828">' + bg + ' için seçilen ' + alanlar.join(', ') + ' UYGUN DEĞİLDİR. Projenin onaylanmama riskine karşı değerleri kontrol etmeniz önerilir.</span>');
    }
    if (potansiyellar.length > 0) {
      const alanlar = potansiyellar.map(function(k) { return k.alan_adi || ''; }).filter(Boolean);
      parcalar.push('<span style="color:#e65100">' + bg + ' için seçilen ' + alanlar.join(', ') + ' dağıtım şirketi kabul kriterleriyle uyumsuzluk gösterebilir. Projenin onaylanmama riskine karşı değerleri kontrol etmeniz önerilir.</span>');
    }
    return parcalar.join('<br><br>');
  }

  // Grup adından sütun başlığını türetir
  function odGrupSutunBasligi(grupAdi) {
    if (grupAdi.includes('KABLO')) return 'Kablo';
    if (grupAdi.includes('KESİCİ')) return 'Kesici';
    if (grupAdi.includes('RÖLE')) return 'Röle';
    if (grupAdi.includes('SAYAÇ')) return 'Sayaç';
    return 'Parametre';
  }

  let html = '<div class="od-baslik">🔍 Ön Değerlendirme Sonuçları</div>';

  Object.keys(gruplar).forEach(function(grupAdi) {
    const kurallar = gruplar[grupAdi];

    // Aynı temel başlık altındaki kuralları birleştir
    const baslikGruplari = {};
    const baslikSirasi = [];
    kurallar.forEach(function(k) {
      const bg = odBaslikGrubu(k.baslik);
      if (!baslikGruplari[bg]) {
        baslikGruplari[bg] = [];
        baslikSirasi.push(bg);
      }
      baslikGruplari[bg].push(k);
    });

    // En az bir görünür (non-atla) kural var mı?
    const hasVisible = baslikSirasi.some(function(bg) {
      return baslikGruplari[bg].some(function(k) { return k.durum !== 'atla'; });
    });
    if (!hasVisible) return;

    // Grup özet durumu
    const hasUygunsuz = baslikSirasi.some(function(bg) {
      return baslikGruplari[bg].some(function(k) { return k.durum === 'uygunsuz'; });
    });
    const hasPotansiyel = baslikSirasi.some(function(bg) {
      return baslikGruplari[bg].some(function(k) { return k.durum === 'potansiyel'; });
    });
    const ozetMetin = hasUygunsuz ? '❌ UYGUNSUZ' : (hasPotansiyel ? '⚠️ RİSK' : '✅ UYGUN');

    html += '<div class="od-grup">';
    html += '<table class="od-tablo">';
    html += '<thead>';
    html += '<tr class="od-tablo-grup-satiri"><th colspan="' + (grupAdi.includes('AKIM HESABI') ? '2' : '4') + '">' + grupAdi + '<span style="float:right">' + ozetMetin + '</span></th></tr>';
    html += '<tr class="od-tablo-sut-satiri">';
    html += '<th>' + odGrupSutunBasligi(grupAdi) + '</th>';
    if (!grupAdi.includes('AKIM HESABI')) {
      html += '<th>Kullanılan</th>';
      html += '<th>Önerilen</th>';
    }
    html += '<th>Değerlendirme</th>';
    html += '</tr>';
    html += '</thead><tbody>';

    baslikSirasi.forEach(function(bg) {
      const list = baslikGruplari[bg];
      // Tüm kurallar atla ise satırı atla
      if (list.every(function(k) { return k.durum === 'atla'; })) return;

      const firstWithData = list.find(function(k) { return k.kullanilan && k.kullanilan !== '—'; }) || list[0];
      const kullanilan = (firstWithData && firstWithData.kullanilan) || '—';
      const onerilen_row = (firstWithData && firstWithData.onerilen_row) || '—';
      const degerlendirme = (list[0] && list[0].custom_degerlendirme !== undefined)
        ? list[0].custom_degerlendirme
        : odDegerlendirmeHtml(grupAdi, bg, list);

      html += '<tr>';
      html += '<td>' + bg + '</td>';
      if (!grupAdi.includes('AKIM HESABI')) {
        html += '<td>' + kullanilan + '</td>';
        html += '<td>' + onerilen_row + '</td>';
      }
      html += '<td>' + degerlendirme + '</td>';
      html += '</tr>';
    });

    html += '</tbody></table>';
    html += '</div>';
  });

  if (html === '<div class="od-baslik">🔍 Ön Değerlendirme Sonuçları</div>') {
    html += '<div style="color:#2e7d32;font-weight:700;padding:14px;background:#e8f5e9;border-radius:8px;margin-top:8px">✅ Tüm kurallar uygun — ön değerlendirme geçti.</div>';
  }

  sonucEl.innerHTML = html;
}

// ANA PDF OLUŞTURMA

async function pdfOlustur() {
  gosterDurum('📥 Veriler okunuyor...', 'info');
  setProgress(5);

  const sema = loadLS(KEY_SEMA) || {};
  const yapi = loadLS(KEY_YAPI) || {};

  gosterDurum('📄 PDF şablonu indiriliyor...', 'info');
  setProgress(10);

  const resp = await fetch(PDF_URL, { cache: 'no-cache' });
  if (!resp.ok) throw new Error('Şablon indirilemedi: HTTP ' + resp.status);
  const templateBytes = await resp.arrayBuffer();

  const { PDFDocument } = window.PDFLib;
  const pdfDoc = await PDFDocument.load(templateBytes);
  const page   = pdfDoc.getPage(0);
  const { width: PW, height: PH } = page.getSize();
  console.log('PDF sayfa boyutu:', PW.toFixed(1), '×', PH.toFixed(1), 'pt');

  // Aktif görevler
  const aktifOgeler = VARSAYILAN_OGELERI.filter(function(og) {
    const ayar  = _ayarlar[og.id] || {};
    const aktif = ayar.aktif != null ? ayar.aktif : og.aktif;
    const kaynak= og.kaynak === 'sema' ? sema : yapi;
    return aktif && kaynak && kaynak[og.alan];
  });

  const toplam = aktifOgeler.length;
  let tamamlanan = 0;

  gosterDurum('🖼️ ' + toplam + ' öğe işlenecek...', 'info');

  for (let i = 0; i < aktifOgeler.length; i++) {
    const og    = aktifOgeler[i];
    const ayar  = _ayarlar[og.id] || {};
    const scale = ayar.scale != null ? ayar.scale : og.scale;
    const posX  = ayar.x     != null ? ayar.x     : og.x;
    const posY  = ayar.y     != null ? ayar.y     : og.y; // üstten aşağı (pt)
    const dpi   = ayar.dpi   != null ? ayar.dpi   : og.dpi;

    const kaynak = og.kaynak === 'sema' ? sema : yapi;
    const veri   = kaynak[og.alan];

    gosterDurum('🖼️ [' + (i+1) + '/' + toplam + '] ' + og.baslik + ' işleniyor...', 'info');

    try {
      let result;

      if (og.tip === 'svg') {
        // SVG → PNG (dpiMul canvas kalite çarpanı)
        result = await svgToPng(veri, og.pxW, og.pxH, dpi);
      } else {
        // HTML → PNG (html2canvas dpi = scale parametresi)
        result = await htmlToPng(veri, og.renderW, dpi);
      }

      // Canvas boyutu dpiMul ile büyütülmüş olduğundan
      // asıl boyutu geri hesapla, sonra scale uygula
      const naturalW = result.w / result.dpiMul;
      const naturalH = result.h / result.dpiMul;
      const finalW   = naturalW * scale;
      const finalH   = naturalH * scale;

      await embedPng(pdfDoc, page, PH, result.dataUrl, posX, posY, finalW, finalH);

      console.log('✅', og.id,
        '→ x:', posX.toFixed(0), 'y:', posY.toFixed(0),
        'w:', finalW.toFixed(0), 'h:', finalH.toFixed(0),
        'dpi×:', dpi
      );
    } catch(e) {
      console.warn('⚠️', og.id, 'yerleştirilemedi:', e.message);
    }

    tamamlanan++;
    setProgress(10 + (tamamlanan / toplam) * 85);
  }

  const finalBytes = await pdfDoc.save();
  pdfBlob = new Blob([finalBytes], { type: 'application/pdf' });
  setProgress(100);
  gosterDurum('✅ PDF hazır! (' + toplam + ' öğe yerleştirildi)', 'tamam');
  return pdfBlob;
}
