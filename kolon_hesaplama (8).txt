// ── Bina görselini writing-mode KULLANMADAN üret (html2canvas uyumlu) ──
function buildBinaSnapshot() {
  var floors = state.floors;
  if (!floors || floors.length === 0) return '';

  var asansor = state.asansor;
  var ortakAlan = state.ortakAlan;
  var istekFloor = state.istekFloor;
  var istekUnit = state.istekUnit;

  var maxU = 1;
  floors.forEach(function(f) { if (f.units.length > maxU) maxU = f.units.length; });

  var cellW = 110, cellH = 80, sideW = 44, labelW = 52, padding = 8;
  var totalCols = maxU;
  var innerW = totalCols * cellW + (totalCols - 1) * 6 + padding * 2;
  var sideCount = (asansor > 0 ? 1 : 0) + (ortakAlan > 0 ? 1 : 0);
  var totalW = sideCount * sideW + innerW + labelW;
  var totalH = floors.length * cellH;

  var html = '<div style="display:inline-flex;align-items:stretch;border:3px double #2c3e50;background:#fff;font-family:Segoe UI,Tahoma,sans-serif;font-size:11px;">';

  // Sol side strips (yatay metin — writing-mode YOK)
  if (sideCount > 0) {
    html += '<div style="display:flex;flex-direction:row;">';
    if (asansor > 0) {
      html += '<div style="width:' + sideW + 'px;border-right:3px double #2c3e50;background:#fff;display:flex;align-items:center;justify-content:center;padding:4px;">';
      html += '<span style="display:inline-block;transform:rotate(-90deg);white-space:nowrap;font-weight:700;font-size:11px;color:#2c3e50;">Asansör x' + asansor + '</span>';
      html += '</div>';
    }
    if (ortakAlan > 0) {
      html += '<div style="width:' + sideW + 'px;border-right:3px double #2c3e50;background:#fff;display:flex;align-items:center;justify-content:center;padding:4px;">';
      html += '<span style="display:inline-block;transform:rotate(-90deg);white-space:nowrap;font-weight:700;font-size:11px;color:#2c3e50;">Ortak Alan x' + ortakAlan + '</span>';
      html += '</div>';
    }
    html += '</div>';
  }

  // Katlar
  html += '<div style="display:flex;flex-direction:column;">';
  for (var i = floors.length - 1; i >= 0; i--) {
    var floor = floors[i];
    var isIstekKat = (istekFloor === i);
    html += '<div style="display:flex;align-items:stretch;border-bottom:' + (i > 0 ? '3px double #2c3e50' : 'none') + ';min-height:' + cellH + 'px;">';

    // Units
    html += '<div style="display:flex;flex-wrap:nowrap;padding:8px;gap:6px;align-items:center;justify-content:center;flex:1;">';
    floor.units.forEach(function(type, uIdx) {
      var isIstek = isIstekKat && uIdx === istekUnit;
      if (isIstek) {
        html += '<div style="min-width:70px;flex:1;min-height:48px;border:2px solid #f39c12;border-radius:4px;display:flex;align-items:center;justify-content:center;font-weight:700;font-size:12px;background:#fef9e7;color:#e67e22;padding:4px 8px;">İSTEK</div>';
      } else {
        html += '<div style="min-width:70px;flex:1;min-height:48px;border:2px double #7f8c8d;border-radius:4px;display:flex;align-items:center;justify-content:center;font-weight:700;font-size:12px;background:#fff;color:#2c3e50;padding:4px 8px;">MEVCUT</div>';
      }
    });
    html += '</div>';

    // Kat etiketi (sağda, yatay OLMAYAN, döndürülmüş span ile)
    html += '<div style="width:' + labelW + 'px;border-left:3px double #2c3e50;display:flex;align-items:center;justify-content:center;background:#fff;">';
    html += '<span style="display:inline-block;transform:rotate(-90deg);white-space:nowrap;font-weight:700;font-size:11px;color:#2c3e50;">' + floor.name + '</span>';
    html += '</div>';

    html += '</div>';
  }
  html += '</div>';
  html += '</div>';

  return html;
}


// Bina görselini Pure SVG olarak oluştur
function _binaSvgOlustur() {
  var floors    = state.floors;
  var asansor   = state.asansor;
  var ortakAlan = state.ortakAlan;
  var istekFloor = state.istekFloor;
  var istekUnit  = state.istekUnit;

  if (!floors || floors.length === 0) return '';

  // Boyutlar
  var stripW    = 38;
  var unitW     = 100;
  var unitH     = 70;
  var unitInnerH = 42;
  var labelW    = 48;
  var padX      = 10;
  var unitGap   = 8;
  var borderW   = 2;

  var maxU = 1;
  floors.forEach(function(f) { if (f.units.length > maxU) maxU = f.units.length; });

  var totalStrips = (asansor > 0 ? 1 : 0) + (ortakAlan > 0 ? 1 : 0);
  var stripsTotalW = totalStrips * stripW;
  var contentW = maxU * unitW + (maxU - 1) * unitGap + padX * 2;
  var totalW = stripsTotalW + contentW + labelW;
  var totalH = floors.length * unitH;

  var svg = '<svg xmlns="http://www.w3.org/2000/svg" width="' + totalW + '" height="' + totalH + '" viewBox="0 0 ' + totalW + ' ' + totalH + '">';

  // Stiller
  svg += '<defs><style>';
  svg += '.bina-border { fill:none; stroke:#2c3e50; stroke-width:' + borderW + '; }';
  svg += '.bina-border-double { fill:none; stroke:#2c3e50; stroke-width:1; }';
  svg += '.bina-text { font-family:Segoe UI,Arial,sans-serif; font-weight:700; fill:#2c3e50; }';
  svg += '.bina-text-istek { font-family:Segoe UI,Arial,sans-serif; font-weight:700; fill:#e67e22; }';
  svg += '.unit-border { fill:#fff; stroke:#7f8c8d; stroke-width:1.5; }';
  svg += '.unit-border-istek { fill:#fef9e7; stroke:#f39c12; stroke-width:2; }';
  svg += '</style></defs>';

  // Dış çerçeve
  svg += '<rect x="1" y="1" width="' + (totalW - 2) + '" height="' + (totalH - 2) + '" fill="#fff" stroke="#2c3e50" stroke-width="2"/>';

  // Side strips (Asansör, Ortak Alan)
  var sx = 0;
  if (asansor > 0) {
    // Strip arka plan
    svg += '<rect x="' + sx + '" y="0" width="' + stripW + '" height="' + totalH + '" fill="#fff"/>';
    // Sağ kenar çizgi
    svg += '<line x1="' + (sx + stripW) + '" y1="0" x2="' + (sx + stripW) + '" y2="' + totalH + '" stroke="#2c3e50" stroke-width="2"/>';
    // Dikey metin
    var cy = totalH / 2;
    svg += '<text x="' + (sx + stripW / 2) + '" y="' + cy + '" transform="rotate(-90,' + (sx + stripW / 2) + ',' + cy + ')" text-anchor="middle" class="bina-text" font-size="11">Asansör x' + asansor + '</text>';
    sx += stripW;
  }
  if (ortakAlan > 0) {
    svg += '<rect x="' + sx + '" y="0" width="' + stripW + '" height="' + totalH + '" fill="#fff"/>';
    svg += '<line x1="' + (sx + stripW) + '" y1="0" x2="' + (sx + stripW) + '" y2="' + totalH + '" stroke="#2c3e50" stroke-width="2"/>';
    var cy2 = totalH / 2;
    svg += '<text x="' + (sx + stripW / 2) + '" y="' + cy2 + '" transform="rotate(-90,' + (sx + stripW / 2) + ',' + cy2 + ')" text-anchor="middle" class="bina-text" font-size="11">Ortak Alan x' + ortakAlan + '</text>';
    sx += stripW;
  }

  var floorsX = sx;

  // Katlar (yukarıdan aşağıya: en üst kat = son index)
  for (var i = floors.length - 1; i >= 0; i--) {
    var rowIdx = floors.length - 1 - i;
    var fy = rowIdx * unitH;
    var floor = floors[i];

    // Kat satırı alt çizgi (son kat hariç)
    if (i > 0) {
      svg += '<line x1="' + floorsX + '" y1="' + (fy + unitH) + '" x2="' + (floorsX + contentW + labelW) + '" y2="' + (fy + unitH) + '" stroke="#2c3e50" stroke-width="2"/>';
    }

    // Birimler
    var numU = floor.units.length;
    var totalUnitsW = numU * unitW + Math.max(0, numU - 1) * unitGap;
    var startUX = floorsX + padX + (contentW - padX * 2 - totalUnitsW) / 2;

    floor.units.forEach(function(type, uIdx) {
      var ux = startUX + uIdx * (unitW + unitGap);
      var uy = fy + (unitH - unitInnerH) / 2;
      var isIstek = (istekFloor === i && istekUnit === uIdx);

      if (isIstek) {
        svg += '<rect x="' + ux + '" y="' + uy + '" width="' + unitW + '" height="' + unitInnerH + '" rx="4" ry="4" class="unit-border-istek"/>';
        svg += '<text x="' + (ux + unitW / 2) + '" y="' + (uy + unitInnerH / 2 + 5) + '" text-anchor="middle" class="bina-text-istek" font-size="12">İSTEK</text>';
      } else {
        svg += '<rect x="' + ux + '" y="' + uy + '" width="' + unitW + '" height="' + unitInnerH + '" rx="4" ry="4" class="unit-border"/>';
        svg += '<text x="' + (ux + unitW / 2) + '" y="' + (uy + unitInnerH / 2 + 5) + '" text-anchor="middle" class="bina-text" font-size="12">MEVCUT</text>';
      }
    });

    // Kat etiketi — sağ taraf dikey ayırıcı + dikey metin
    var labelX = floorsX + contentW;
    svg += '<line x1="' + labelX + '" y1="' + fy + '" x2="' + labelX + '" y2="' + (fy + unitH) + '" stroke="#2c3e50" stroke-width="2"/>';
    var labelCY = fy + unitH / 2;

  function distributeUnits() {
    var totalFloors = state.katSayisi + 1;
    var base = Math.floor(state.toplamBagimsiz / totalFloors);
    var remainder = state.toplamBagimsiz % totalFloors;
    var newFloors = [];
    for (var i = 0; i < totalFloors; i++) {
      var cnt = base + (i >= totalFloors - remainder ? 1 : 0);
      var units = [];
      for (var u = 0; u < cnt; u++) units.push('mevcut');
      newFloors.push({ name: i === 0 ? 'Zemin Kat' : i + '. Kat', units: units });
    }
    if (state.istekFloor !== null && state.istekUnit !== null) {
      if (state.istekFloor < newFloors.length &&
          state.istekUnit < newFloors[state.istekFloor].units.length) {
        newFloors[state.istekFloor].units[state.istekUnit] = 'istek';
      } else {
        state.istekFloor = null; state.istekUnit = null;
      }
    }
    state.floors = newFloors;
  }

  function setIstekOrtak(floorIdx, unitIdx) {
    if (state.istekFloor !== null && state.istekFloor >= 0 &&
        state.istekUnit !== null &&
        state.istekFloor < state.floors.length &&
        state.istekUnit < state.floors[state.istekFloor].units.length) {
      state.floors[state.istekFloor].units[state.istekUnit] = 'mevcut';
    }
    state.istekFloor = floorIdx; state.istekUnit = unitIdx;
    closeAllMenus(); _binaIstekTetikleme = true; renderBuilding(); _binaIstekTetikleme = false;
    var label = unitIdx < state.asansor ? 'Asansör' : 'Ortak Alan';
    showToast('     ' + label + ' İSTEK olarak işaretlendi.');
  }

  function setIstekAuto(floorIdx) {
    if (state.istekFloor === -1) { state.istekFloor = null; state.istekUnit = null; }
    if (state.istekFloor !== null && state.istekUnit !== null &&
        state.istekFloor < state.floors.length &&
        state.istekUnit < state.floors[state.istekFloor].units.length) {
      state.floors[state.istekFloor].units[state.istekUnit] = 'mevcut';
    }
    var floor = state.floors[floorIdx], sel = 0;
    for (var u = 0; u < floor.units.length; u++) { if (floor.units[u] === 'mevcut') { sel = u; break; } }
    floor.units[sel] = 'istek'; state.istekFloor = floorIdx; state.istekUnit = sel;
    closeAllMenus(); _binaIstekTetikleme = true; renderBuilding(); _binaIstekTetikleme = false;
    showToast('🔶 ' + floor.name + ' - Bağımsız ' + (sel + 1) + ' İSTEK olarak işaretlendi.');
  }

  function addUnit(floorIdx) {
    if (getActiveUnitCount() >= state.toplamBagimsiz) { showToast('⚠️ Toplam bağımsız sayısına ulaşıldı!'); closeAllMenus(); return; }
    state.floors[floorIdx].units.push('mevcut'); closeAllMenus(); renderBuilding();
    showToast('✅ ' + state.floors[floorIdx].name + ' katına bağımsız eklendi.');
  }

  function removeUnit(floorIdx) {
    var floor = state.floors[floorIdx];
    if (floor.units.length === 0) { showToast('⚠️ Bu katta silinecek bağımsız yok.'); closeAllMenus(); return; }
    if (state.istekFloor === floorIdx && state.istekUnit >= floor.units.length - 1) {
      state.istekFloor = null; state.istekUnit = null;
    }
    floor.units.pop(); closeAllMenus(); renderBuilding();
    showToast('🗑️ ' + floor.name + ' katından bağımsız silindi.');
  }

    }
    for (var i = 1; i <= state.toplamBagimsiz; i++) {
      var optB = document.createElement('option');
      optB.value = 'bagimsiz_' + i;
      optB.textContent = i + ' Nolu Bağımsız';
      sel.appendChild(optB);
    }
    if (state.istekFloor === -1 && state.istekUnit !== null) {
      sel.value = state.istekUnit < state.asansor ? 'asansor' : 'ortak';
    } else if (state.istekFloor !== null && state.istekFloor >= 0 && state.istekUnit !== null) {
      var globalN = 0;
      var found = false;
      for (var fi = 0; fi < state.floors.length && !found; fi++) {
        for (var ui = 0; ui < state.floors[fi].units.length && !found; ui++) {
          globalN++;
          if (fi === state.istekFloor && ui === state.istekUnit) {
            sel.value = 'bagimsiz_' + globalN;
            found = true;
          }
        }
      }
    }
  }

  function hesaplaKoordinatlar(floors, istekFloor, istekOrtakMi) {
    var katSayisi = floors.length;
    var topY = 30;
    var istekEkYukseklik = 20;
    for (var iter = 0; iter < 2; iter++) {
      var ky = new Array(katSayisi);
      var y = topY;
      for (var i = katSayisi - 1; i >= 0; i--) {
        ky[i] = y;
        y += (i === istekFloor && !istekOrtakMi) ? KAT_YUKSEKLIGI + istekEkYukseklik : KAT_YUKSEKLIGI;
      }
      if (istekFloor !== null && !istekOrtakMi) {
        var istekH       = KAT_YUKSEKLIGI + istekEkYukseklik;
        var istekKatUst  = ky[istekFloor];
        var istekKatAlt  = istekKatUst + istekH;
        var dtUstY_istek = istekKatUst + ISTEK_DT_UST_OFFSET;
        var dtAltY_istek = dtUstY_istek + DT.h;
        var sayacGirisAlt = dtAltY_istek + SAYAC_GIRIS_DERINLIK;
        var gereken      = sayacGirisAlt + ISTEK_PAY - istekKatAlt;
        istekEkYukseklik = Math.max(0, gereken);
      }
    }
    var katUstYler = new Array(katSayisi);
    var y2 = topY;
    for (var i2 = katSayisi - 1; i2 >= 0; i2--) {
      katUstYler[i2] = y2;
      y2 += (i2 === istekFloor && !istekOrtakMi) ? KAT_YUKSEKLIGI + istekEkYukseklik : KAT_YUKSEKLIGI;
    }
    return {
      katUstYler: katUstYler,
      adpKatUst: y2,
      istekEkYukseklik: istekEkYukseklik,
      katH: function (idx) { return (idx === istekFloor && !istekOrtakMi) ? KAT_YUKSEKLIGI + istekEkYukseklik : KAT_YUKSEKLIGI; },
      katAltY:  function (idx) { return katUstYler[idx] + this.katH(idx); },
      katOrtaY: function (idx) { return katUstYler[idx] + this.katH(idx) / 2; },
      dtUstY:   function (idx) {
        if (idx === istekFloor && !istekOrtakMi) return katUstYler[idx] + ISTEK_DT_UST_OFFSET;
        return this.katOrtaY(idx) - DT.h / 2;
      },
      dtAltY: function (idx) { return this.dtUstY(idx) + DT.h; }
    };
  }

      s+='<rect x="'+ix+'" y="'+gy+'" width="'+iw+'" height="'+gozH+'" class="p-goz"/>';
      var buGozdeki=gozDagitim[g];
      if(buGozdeki>0){
        var sayacW=PANO_SAYAC.width;
        var sayacH=Math.min(PANO_SAYAC.height, gozH);
        var sayacY=gy+(gozH-sayacH)/2;
        var oncekiToplam=0;
        for(var k=0;k<g;k++) oncekiToplam+=gozDagitim[k];
        for(var si=0;si<buGozdeki;si++){
          var sx=ix+si*(iw/D.SAYAC_PER_GOZ)+(iw/D.SAYAC_PER_GOZ-sayacW)/2;
          s+=drawPanoSayac(sx, sayacY, sayacW, sayacH, startIndex+oncekiToplam+si+1);
        }
      }
    }
    var termY=iy+ih-D.PANO.terminalH;
    s+='<line x1="'+ix+'" y1="'+termY+'" x2="'+(ix+iw)+'" y2="'+termY+'" class="p-inner"/>';
    s+='<rect x="'+(ix+1)+'" y="'+(termY+6)+'" width="9" height="'+(D.PANO.terminalH-6)+'" class="p-term"/>';
    s+='<rect x="'+(ix+iw-10)+'" y="'+(termY+6)+'" width="9" height="'+(D.PANO.terminalH-6)+'" class="p-term"/>';
    for(var row=0;row<3;row++){
      var ly=termY+12+row*19;
      s+='<line x1="'+(ix+10)+'" y1="'+ly+'" x2="'+(ix+iw-10)+'" y2="'+ly+'" class="p-inner"/>';
      s+='<circle cx="'+(ix+5.5)+'" cy="'+(ly+(row===0?6:row===1?-2:-10))+'" r="0.7" class="p-dot"/>';
      s+='<circle cx="'+(ix+iw-5.5)+'" cy="'+(ly+(row===0?6:row===1?-2:-10))+'" r="0.7" class="p-dot"/>';
    }
    return s;
  }

    s+='<rect x="'+scrX+'" y="'+(y+31.4)+'" width="'+scrW+'" height="'+scrH+'" class="p-inner"/>';

  function drawDuvarPano(x, y, startIndex, sayacCount, aktifGoz, gozW, tmsVar) {
    var s='';
    var D=DUVAR;
    var sema = loadSemaVerileri();
    var kolonSigortaTip = (sema && sema.kolonSigortaTip) || '';
    var sigortaLabel = (kolonSigortaTip === 'K.Otomat') ? 'K.Otomat' : 'TMŞ';
    var innerW=aktifGoz<=2 ? gozW+D.divW+D.sayacColW : gozW+D.divW+D.sayacColW+gozW;
    var outerW=innerW+D.pad*2;
    var outerH=D.gozH*2+D.pad*2;
    var innerH=D.gozH*2;
    var ix=x+D.pad, iy=y+D.pad;
    s+='<rect x="'+x+'" y="'+y+'" width="'+outerW+'" height="'+outerH+'" class="p-outer"/>';
    s+='<rect x="'+ix+'" y="'+iy+'" width="'+innerW+'" height="'+innerH+'" class="p-inner"/>';
    var divX=ix+gozW;
    s+='<rect x="'+divX+'" y="'+iy+'" width="'+D.divW+'" height="'+innerH+'" class="p-inner"/>';
    var scX=divX+D.divW;
    s+=drawDuvarSabitSayacUst(scX, iy, D.sayacColW, D.gozH);
    s+=drawDuvarSabitSayacAlt(scX, iy+D.gozH, D.sayacColW, D.gozH);
    if(tmsVar){
      var tmsLabelX=divX+(D.divW+D.sayacColW)/2;
      s+='<text x="'+(tmsLabelX+5)+'" y="'+(iy+innerH/2+60)+'" text-anchor="middle" class="p-txt">'+sigortaLabel+'</text>';
    }
    var solX=ix, ustY=iy, altY=iy+D.gozH;
    s+='<rect x="'+solX+'" y="'+ustY+'" width="'+gozW+'" height="'+D.gozH+'" class="p-goz"/>';
    s+='<rect x="'+solX+'" y="'+altY+'" width="'+gozW+'" height="'+D.gozH+'" class="p-goz"/>';
    var sagX;
    if(aktifGoz>2){
      sagX=scX+D.sayacColW;
      s+='<rect x="'+sagX+'" y="'+ustY+'" width="'+gozW+'" height="'+D.gozH+'" class="p-goz"/>';
      s+='<rect x="'+sagX+'" y="'+altY+'" width="'+gozW+'" height="'+D.gozH+'" class="p-goz"/>';
    }
    var gozPositions=[{gx:solX,gy:altY},{gx:solX,gy:ustY}];
    if(aktifGoz>2){ gozPositions.push({gx:sagX,gy:altY}); gozPositions.push({gx:sagX,gy:ustY}); }
    var gozDagitim=esitDagit(sayacCount, aktifGoz);
    var sayacNo=startIndex;
    for(var g=0;g<aktifGoz;g++){
      var buGozdeki=gozDagitim[g];
      if(buGozdeki>0){
        var gp=gozPositions[g];
        var sayacW=PANO_SAYAC.width;
        var sayacH=Math.min(PANO_SAYAC.height, D.gozH);
        var sayacY=gp.gy+(D.gozH-sayacH)/2;
        for(var si=0;si<buGozdeki;si++){
          var sx=gp.gx+si*(gozW/buGozdeki)+(gozW/buGozdeki-sayacW)/2;
          s+=drawPanoSayac(sx, sayacY, sayacW, sayacH, sayacNo+1);
          sayacNo++;
        }
      }
    }
    return s;
  }

  function renderDikiliPano(tesisatSayisi, sigortaVar) {
    var D=DIKILI;
    var sema = loadSemaVerileri();
    var kolonSigortaTip = (sema && sema.kolonSigortaTip) || '';
    var sigortaLabel = (kolonSigortaTip === 'K.Otomat') ? 'K.Otomat' : 'TMŞ';
    var panoSayisi=Math.ceil(tesisatSayisi/D.SAYAC_PER_PANO);
    $('panoInfoBar').innerHTML=
      '<b>Tip:</b> Dikili &nbsp;|&nbsp; <b>Toplam Tesisat:</b> '+tesisatSayisi+
      ' &nbsp;|&nbsp; <b>Pano Sayısı:</b> '+panoSayisi+
      ' &nbsp;|&nbsp; <b>'+sigortaLabel+':</b> '+(sigortaVar?'Var':'Yok')+
      ' &nbsp;|&nbsp; <b>Sayaç/Pano:</b> '+D.SAYAC_PER_PANO;
    var tmsW=sigortaVar?D.TMS.width:0;
    var totalW=tmsW+panoSayisi*D.PANO.width+140;
    var totalH=D.PANO.height+D.PANO.marginTop+50;
    var svg=panoSvgOpen(totalW, totalH);
    var baseX=20, baseY=D.PANO.marginTop;
    if(sigortaVar) svg+=panoOlcuYatay(baseX, baseY-15, D.TMS.width, D.TMS.genislikCm);
    for(var p=0;p<panoSayisi;p++) svg+=panoOlcuYatay(baseX+tmsW+p*D.PANO.width, baseY-15, D.PANO.width, D.PANO.genislikCm);
    if(sigortaVar){
      svg+='<text x="'+(baseX-5+D.TMS.width/2)+'" y="'+(baseY+410)+'" text-anchor="middle" class="p-txt">'+sigortaLabel+'</text>';
      svg+=drawDikiliTMS(baseX, baseY);
    }
    var sayacIndex=0;
    for(var p=0;p<panoSayisi;p++){
      var px=baseX+tmsW+p*D.PANO.width;
      var kalan=tesisatSayisi-sayacIndex;
      var bu=Math.min(D.SAYAC_PER_PANO, kalan);
      svg+=drawDikiliPano(px, baseY, sayacIndex, bu);
      sayacIndex+=bu;
    }
    svg+=panoOlcuDikey(baseX+tmsW+panoSayisi*D.PANO.width+30, baseY, D.PANO.height, '210 cm');
    svg+='</svg>';
    $('panoSvgContainer').innerHTML=svg;
  }

  function renderDuvarPano(tesisatSayisi, sigortaVar) {
    var D=DUVAR;
    var sema = loadSemaVerileri();
    var kolonSigortaTip = (sema && sema.kolonSigortaTip) || '';
    var sigortaLabel = (kolonSigortaTip === 'K.Otomat') ? 'K.Otomat' : 'TMŞ';
    var aktifGoz=(tesisatSayisi<=8)?2:4;
    var perGozMax=Math.ceil(tesisatSayisi/aktifGoz);
    var gozW=perGozMax*PANO_SAYAC.width;
    var innerW=aktifGoz<=2 ? gozW+D.divW+D.sayacColW : gozW+D.divW+D.sayacColW+gozW;
    var outerW=innerW+D.pad*2;
    var outerH=D.gozH*2+D.pad*2;
    var genislikCm;
    if(aktifGoz<=2){
      var baseHalfOuterW=D.pad*2+D.baseGozW+D.divW+D.sayacColW;
      genislikCm=Math.round((outerW/baseHalfOuterW)*150);
    } else {
      var baseFullOuterW=D.pad*2+D.baseGozW*2+D.divW+D.sayacColW;
      genislikCm=Math.round((outerW/baseFullOuterW)*210);
    }
    $('panoInfoBar').innerHTML=
      '<b>Tip:</b> Duvar &nbsp;|&nbsp; <b>Toplam Tesisat:</b> '+tesisatSayisi+
      ' &nbsp;|&nbsp; <b>Aktif Göz:</b> '+aktifGoz+
      ' &nbsp;|&nbsp; <b>Max Sayaç/Göz:</b> '+perGozMax+
      ' &nbsp;|&nbsp; <b>'+sigortaLabel+':</b> '+(sigortaVar?'Var':'Yok')+
      ' &nbsp;|&nbsp; <b>Pano:</b> '+genislikCm+' × 120 cm';
    var totalW=outerW+100;
    var totalH=outerH+D.marginTop+80;
    var svg=panoSvgOpen(totalW, totalH);
    var baseX=20, baseY=D.marginTop;
    svg+=drawDuvarPano(baseX, baseY, 0, tesisatSayisi, aktifGoz, gozW, sigortaVar);
    svg+=panoOlcuYatay(baseX, baseY-15, outerW, genislikCm+' cm');
    svg+=panoOlcuDikey(baseX+outerW+15, baseY, outerH, '120 cm');
    svg+='</svg>';
    $('panoSvgContainer').innerHTML=svg;
  }

    function renderPano() {
    var tesisatSayisi=state.totalTesisat;

  }
